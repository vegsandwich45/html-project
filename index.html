# Z-Test (Sample Mean vs Population Mean)


xbar=14.6

mu0=15.4

sigma=2.5

n=35


z=(xbar-mu0)/(sigma/sqrt(n))


alpha=0.05


zhalfalpha=qnorm(1-(alpha/2))


pval=2*pnorm(z)


if(pval>alpha){

print("Accept Null hypothesis")

}else{

print("Reject Null hypothesis")

}

# Z-Test (Two Sample Proportions)


p1=0.20

n1=900

z=(p1-p2)/sqrt(P*Q*((1/n1)+(1/n2)))

if(z<=zalpha){

print("Reject Null hypothesis")

}

# Chi-Square Test (Goodness of Fit)


n=5

alpha=0.05

N=256


P=0.5


x=c(0:n)


obf=c(5,35,75,84,45,12)


exf=(dbinom(x,n,P)*256)


chisq<-sum((obf-exf)^2/exf)


cv=chisq


tv=qchisq(1-alpha,n)


if(cv <= tv){

print("Accept H0/Fit is good")

}else{

print("Reject H0/Fit is not good")

}
